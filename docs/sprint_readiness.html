import pandas as pd import json # Load CSV df = pd.read_csv("jira_data.csv") # Clean column names df.columns = [col.strip() for col in df.columns] # Filter for story tickets that are in a future sprint story_df = df[ (df["Issue Type"] == "Story") & (df["Status"].isin(["To Do", "Ready for Development", "New", "Grooming"])) & (df["Sprint"].notnull()) & (~df["Sprint"].str.contains("state=ACTIVE", na=False)) & (df["Status"] != "Done") ] # Map components to teams team_names = [ "Engineering - Product", "Engineering - Platform", "Engineering - AI Ops", "Design", "Data Science" ] sprint_data = {} for team in team_names: team_df = story_df[story_df["Components"].str.contains(team, na=False)] status_counts = team_df["Status"].value_counts().to_dict() ready_count = sum([status_counts.get(status, 0) for status in ["To Do", "Ready for Development"]]) total_count = sum(status_counts.values()) readiness = (ready_count / total_count) * 100 if total_count > 0 else 0 if readiness >= 80: stoplight = "green" elif readiness >= 50: stoplight = "yellow" else: stoplight = "red" sprint_data[team] = { "status_counts": status_counts, "ready_count": ready_count, "total_count": total_count, "readiness_percent": round(readiness, 2), "stoplight": stoplight } # Final JSON structure output = { "sprint_readiness": sprint_data } # Write to render_data.json (append or update existing) with open("docs/render_data.json", "r") as infile: full_data = json.load(infile) full_data["sprint_readiness"] = output["sprint_readiness"] with open("docs/render_data.json", "w") as outfile: json.dump(full_data, outfile, indent=2) print("✅ Sprint readiness data updated successfully.")
